input { /*去掉文本框选择线*/ outline: none; } /* 不可操作的文本框变为红色 */ input:disabled { background-color: #ddd; } /* 可操作的文本框变为绿色 */ input:enabled { background-color: green; } /* 选择的单选按钮文字变为绿色 */ input:checked+label { color: green; } /* 将必填框的的边框改为蓝色 */ input:required { border: solid 2px blue; } /* 将非必填框的的边框改为黄色 */ input:optional { border: solid 2px yellow; }
